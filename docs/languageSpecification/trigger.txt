Trigger {
  type TriggerType
  trigger Choice {
    ZoneChangeTrigger,
    PhaseTrigger,
    BattleOpponentReversedTrigger,
    TriggerRevealTrigger,
    OnPlayTrigger,
    OnAttackTrigger,
    OtherTrigger,
    Nothing
  }
}

enum TriggerType {
  onZoneChange = 1,
  onPlay = 2,
  onReverse = 3,
  onAttack = 4,
  onBattleOpponentReversed = 5,
  onBackupOfThis = 6,
  onTriggerReveal = 7,
  onPhaseEvent = 8
  onEndOfThisCardsAttack = 9,
  onOppCharPlacedByStandbyTriggerReveal = 10,
  onEndOfThisTurn = 11,
  
  otherTrigger = 255
}

ZoneChangeTrigger {
  target Array of Target,  // targets are ORed
  from Zone,
  to Zone
}

PhaseTrigger {
  state PhaseState
  phase Phase
  owner Player
}  

TriggerRevealTrigger {
  card Card
}

OnPlayTrigger {
  target Target
}

OnAttackTrigger {
  target Target
}

BattleOpponentReversedTrigger {
  card Card
}

OtherTrigger {
  cardCode String
}
