Trigger {
  type TriggerType
  trigger Choice {
    ZoneChangeTrigger,
    PhaseTrigger,
    StateChangeTrigger,
    TriggerRevealTrigger,
    OnPlayTrigger,
    OnAttackTrigger,
    OtherTrigger,
    Nothing
  }
}

enum TriggerType {
  onZoneChange = 1,
  onPlay = 2,
  onStateChanged = 3,
  onAttack = 4,
  onBackupOfThis = 5,
  onTriggerReveal = 6,
  onPhaseEvent = 7,
  onEndOfThisCardsAttack = 8,
  onOppCharPlacedByStandbyTriggerReveal = 9,
  onEndOfThisTurn = 10,
  
  otherTrigger = 255
}

ZoneChangeTrigger {
  target Array of Target,  // targets are ORed
  from Zone,
  to Zone
}

PhaseTrigger {
  state PhaseState,
  phase Phase,
  owner Player
}  

TriggerRevealTrigger {
  card Card
}

OnPlayTrigger {
  target Target
}

OnAttackTrigger {
  target Target
}

StateChangeTrigger {
  state State,
  target Target
}

OtherTrigger {
  cardCode String
}
